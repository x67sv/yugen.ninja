empire➜  nuclei  ᐅ  cat directory-scanner.yaml
id: DirectoryScanner # yaml ID

info:
  name: Directory Scanner 
  author: Yugen
  severity: low # sonuclarin onem derecesi

requests:
  - method: GET
    path:
      - "{{BaseURL}}/{{fuzz}}"

    payloads:
        fuzz: Wordlists/SecLists/Discovery/Web-Content/big.txt # wordlistin yolu

    threads: 50 # tek seferde 50 request
    matchers:
      - type: status # status 301,200 veya 403 donerse low severity de ekrana sonucu bas
        status:
          - 301
          - 200
          - 403
